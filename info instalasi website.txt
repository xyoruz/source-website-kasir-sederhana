-- 1. Buat Database (Jika belum ada)
CREATE DATABASE IF NOT EXISTS db_xailla;
USE db_xailla;

-- 2. Tabel Admin (Untuk Login)
CREATE TABLE IF NOT EXISTS admin (
    id INT(11) NOT NULL AUTO_INCREMENT,
    username VARCHAR(50) NOT NULL,
    password VARCHAR(255) NOT NULL,
    PRIMARY KEY (id)
);

-- Masukkan akun default (User: admin, Pass: admin)
INSERT INTO admin (username, password) VALUES ('admin', 'admin');

-- 3. Tabel Member (Data Pelanggan)
CREATE TABLE IF NOT EXISTS member (
    id INT(11) NOT NULL AUTO_INCREMENT,
    nama_member VARCHAR(100) NOT NULL,
    PRIMARY KEY (id)
);

-- 4. Tabel Produk (Stok & Harga)
CREATE TABLE IF NOT EXISTS produk (
    id INT(11) NOT NULL AUTO_INCREMENT,
    nama_barang VARCHAR(255) NOT NULL,
    stok INT(11) NOT NULL DEFAULT 0,
    modal INT(11) NOT NULL DEFAULT 0,
    member INT(11) NOT NULL DEFAULT 0,
    ecer INT(11) NOT NULL DEFAULT 0,
    PRIMARY KEY (id)
);

-- 5. Tabel Laporan (Riwayat Penjualan Kasir)
CREATE TABLE IF NOT EXISTS laporan (
    id INT(11) NOT NULL AUTO_INCREMENT,
    id_harian VARCHAR(50) NOT NULL,
    nama_barang VARCHAR(255) NOT NULL,
    tipe_pembeli VARCHAR(20) NOT NULL, -- Ecer atau Member
    harga_jual INT(11) NOT NULL,
    untung INT(11) NOT NULL,
    modal_asli INT(11) NOT NULL,
    stok_sisa INT(11) NOT NULL,
    pembeli VARCHAR(100) NOT NULL,
    tanggal DATETIME NOT NULL,
    PRIMARY KEY (id)
);

-- 6. Tabel History Faktur (Riwayat Pembelian Barang/Kulakan)
CREATE TABLE IF NOT EXISTS history_faktur (
    id INT(11) NOT NULL AUTO_INCREMENT,
    no_faktur VARCHAR(50) NOT NULL,
    suplayer VARCHAR(100) NOT NULL,
    tgl_nota DATETIME NOT NULL,
    nama_barang VARCHAR(255) NOT NULL,
    stok_masuk INT(11) NOT NULL,
    modal INT(11) NOT NULL,
    member INT(11) NOT NULL,
    ecer INT(11) NOT NULL,
    tanggal_input DATETIME NOT NULL,
    PRIMARY KEY (id)
);
